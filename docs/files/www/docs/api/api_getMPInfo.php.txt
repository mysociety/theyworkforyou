function api_getMPinfo_id($id) {
    if (!ctype_digit($id)) {
        api_error('Unknown person ID');
        return;
    }

    $output = _api_getMPsInfo_id($id);
    if ($output) {
        if ($output[0]) {
            api_output($output[0][$id], $output[1]);
        } else {
            api_error('Unknown field');
        }
    } else {
        api_error('Unknown person ID');
    }
}
